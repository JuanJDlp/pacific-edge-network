! ==================== CONFIGURACIÓN BÁSICA ====================
hostname SW-CORE-BONGO

! Contraseña de enable (modo privilegiado)
enable secret password

! Cifrar todas las contraseñas en texto plano del running-config
service password-encryption

! ==================== USUARIO LOCAL ====================
! Requerido para "login local" en VTY (Telnet y SSH)
username user privilege 15 secret password

! ==================== SSH (versión 2) ====================
! El Catalyst 2960 necesita domain-name para generar el par de claves RSA
ip domain-name comunitaria.local

! Genera clave RSA de 2048 bits — habilita SSH automáticamente
! En algunas versiones de IOS pide confirmación; presionar Enter para aceptar
crypto key generate rsa modulus 2048

ip ssh version 2
ip ssh time-out 60
ip ssh authentication-retries 3

! Deshabilitar servidor web (no necesario, reduce superficie de ataque)
no ip http server
no ip http secure-server

! ==================== ACCESO REMOTO (SSH + TELNET) ====================
line vty 0 15
 login local
 transport input ssh telnet
 exec-timeout 30 0
exit

! ==================== CREACIÓN DE VLANs ====================
vlan 10
 name Gestion
exit
vlan 20
 name Servidores
exit
vlan 30
 name Clientes
exit
vlan 40
 name EnlaceCocalito
exit

! ==================== SVI DE GESTIÓN ====================
! IP de gestión del switch, accesible desde VLAN 10 (Mini PC: 192.168.10.1)
interface vlan 10
 description Gestion_SW-CORE-BONGO
 ip address 192.168.10.2 255.255.255.0
 no shutdown
exit

! Gateway por defecto del switch (el Mini PC actúa como router)
ip default-gateway 192.168.10.1

! ==================== SPANNING-TREE ====================
spanning-tree mode rapid-pvst

! ==================== CONFIGURACIÓN DE PUERTOS ====================

! Puertos 1-2: RPi5 y servidores — VLAN 20 access
interface FastEthernet0/1
 description RPi5_Servicios
 switchport mode access
 switchport access vlan 20
 spanning-tree portfast
 no shutdown
exit

interface FastEthernet0/2
 description Servidor_VLAN20
 switchport mode access
 switchport access vlan 20
 spanning-tree portfast
 no shutdown
exit

! Puerto 3: Radio Enlace 900 MHz hacia Cocalito — VLAN 10 access
! (el enlace de radio es un dispositivo de gestión/infraestructura)
interface FastEthernet0/3
 description RadioEnlace_900MHz_Cocalito
 switchport mode access
 switchport access vlan 10
 no shutdown
exit

! Puerto 4: Access Point WiFi — trunk VLAN 20 (SSID servicios) y VLAN 30 (SSID clientes)
interface FastEthernet0/4
 description AP_PacificEdge_WiFi
 switchport mode trunk
 switchport trunk allowed vlan 20,30
 switchport trunk native vlan 1
 no shutdown
exit

! Puerto 24: Uplink al Mini PC — trunk con todas las VLANs
! El Mini PC crea sub-interfaces VLAN etiquetadas; native VLAN 1 queda sin IP
interface FastEthernet0/24
 description Uplink_MiniPC_Router
 switchport mode trunk
 switchport trunk allowed vlan 10,20,30,40
 switchport trunk native vlan 1
 no shutdown
exit

! ==================== HORA ====================
clock timezone COT -5

end

write memory

!

!

!

!   ping 192.168.10.1
!     → Debe responder (Mini PC, gateway)
!
!   show ssh
